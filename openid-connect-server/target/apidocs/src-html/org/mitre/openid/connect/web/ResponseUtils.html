001package org.mitre.openid.connect.web;
002
003import org.springframework.http.HttpStatus;
004
005public class ResponseUtils {
006        
007        public static <T extends Object> ResponseEntity<T> renderSuccess(T object, HttpStatus status) {
008                return ResponseUtils.renderSuccess(object, status, 0);
009        }
010
011        public static <T extends Object> ResponseEntity<T> renderSuccess(T object, HttpStatus status, int total) {
012                ResponseEntity<T> responseEntity = new ResponseEntity<T>();
013                responseEntity.setSuccess(true);
014                responseEntity.setBody(object);
015                responseEntity.setAuthenticated(true);
016                responseEntity.setStatusCode(status);
017                responseEntity.setTotal(total);
018                return responseEntity;
019        }
020
021        public static <T extends Object> ResponseEntity<T> renderError(ResponseEntity<T> responseEntity, Throwable throwable) {
022                responseEntity.setSuccess(false);
023                if (throwable instanceof NullPointerException) {
024                        responseEntity.setErrorMessage("Unhandled Server Exception");
025                } else {
026                        responseEntity.setErrorMessage(throwable.getMessage());
027                }
028                responseEntity.setStatusCode(HttpStatus.BAD_REQUEST);
029                responseEntity.setAuthenticated(true);
030                return responseEntity;
031        }
032        
033        public static <T extends Object> ResponseEntity<T> renderCustomError(ResponseEntity<T> responseEntity, String errorMessage) {
034                responseEntity.setSuccess(false);
035                responseEntity.setAuthenticated(true);
036                responseEntity.setErrorMessage(errorMessage);
037                responseEntity.setStatusCode(HttpStatus.BAD_REQUEST);
038                return responseEntity;
039        }
040        
041        public static <T extends Object> ResponseEntity<T> renderCustomError(ResponseEntity<T> responseEntity, String errorMessage,boolean authenticated) {
042                responseEntity.setSuccess(false);
043                responseEntity.setAuthenticated(authenticated);
044                responseEntity.setErrorMessage(errorMessage);
045                responseEntity.setStatusCode(HttpStatus.BAD_REQUEST);
046                return responseEntity;
047        }
048}
