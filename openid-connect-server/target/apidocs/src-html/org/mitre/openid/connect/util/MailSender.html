001package org.mitre.openid.connect.util;
002import org.springframework.mail.SimpleMailMessage;
003import org.springframework.mail.javamail.JavaMailSender;
004
005public class MailSender implements Runnable {
006
007        private String emailTo;
008        private String body;
009        private String subject;
010        private JavaMailSender mailSender;
011
012        public MailSender(String emailTo, String subject, String body, JavaMailSender mailSender) {
013                this.emailTo = emailTo;
014                this.subject = subject;
015                this.body = body;
016                this.mailSender = mailSender;
017        }
018
019        public void run() {
020                SimpleMailMessage email = new SimpleMailMessage();
021                email.setTo(emailTo);
022                email.setSubject(subject);
023                email.setText(body);
024                mailSender.send(email);
025        }
026
027}
