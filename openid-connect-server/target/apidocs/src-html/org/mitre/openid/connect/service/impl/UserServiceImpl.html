001package org.mitre.openid.connect.service.impl;
002
003import org.mitre.openid.connect.model.User;
004import org.mitre.openid.connect.repository.UserRepository;
005import org.mitre.openid.connect.service.UserService;
006import org.springframework.beans.factory.annotation.Autowired;
007import org.springframework.stereotype.Service;
008
009@Service
010public class UserServiceImpl implements UserService{
011        
012        @Autowired
013        private UserRepository userRepository;
014        
015/*      for new user creation and while updating the password or enabled  */
016        @Override
017        public User saveOrUpdateUser(User user) {
018                System.out.println("UserServiceImpl.createUser "+user.getUsername());
019                getUserByUsername(user.getUsername());
020                return userRepository.saveOrUpdateUser(user);
021        }
022
023        @Override
024        public User getUserByUsername(String username) {
025                System.out.println("UserServiceImpl.getUserByUsername "+username);
026                return userRepository.getUserByUsername(username);
027        }
028
029}
